<script>
    let currentTotal = 10;
    let currentFilled = 7;

    function renderStamps() {
        const grid = document.getElementById('stampsGrid');
        grid.innerHTML = '';
        const primaryColor = document.getElementById('primaryColor').value;
        
        for (let i = 0; i < currentTotal; i++) {
            const isFilled = i < currentFilled;
            const stamp = document.createElement('div');
            stamp.className = 'aspect-square rounded-full flex items-center justify-center stamp-stagger border transition-all duration-300';
            stamp.style.animationDelay = (i * 0.05) + 's';
            
            if (isFilled) {
                stamp.classList.add('bg-primary-container', 'border-transparent');
                stamp.style.backgroundColor = primaryColor;
                stamp.style.boxShadow = `0 0 12px ${primaryColor}66`;
                stamp.innerHTML = `<span class="material-symbols-outlined text-on-primary-container text-sm" style="font-variation-settings: 'FILL' 1;">verified</span>`;
            } else {
                stamp.classList.add('bg-white/5', 'border-white/10');
                stamp.innerHTML = `<div class="w-1.5 h-1.5 rounded-full bg-white/20"></div>`;
            }
            grid.appendChild(stamp);
        }
    }

    function updateColorPreview(val, target) {
        const card = document.getElementById('cardPreview');
        
        if (target === 'primary') {
            document.getElementById('primaryHex').innerText = val.toUpperCase();
            renderStamps();
        } else if (target === 'secondary') {
            document.getElementById('secondaryHex').innerText = val.toUpperCase();
        } else if (target === 'bg') {
            card.style.backgroundColor = val;
            document.getElementById('bgHex').innerText = val.toUpperCase();
        }
    }

    function updateStamps(val) {
        currentTotal = parseInt(val) || 10;
        renderStamps();
    }

    // Initial Render
    renderStamps();

    // Micro-interaction for file uploaders
    document.querySelectorAll('.luminescent-border').forEach(el => {
        el.addEventListener('mousedown', () => {
            el.classList.add('scale-95');
        });
        el.addEventListener('mouseup', () => {
            el.classList.remove('scale-95');
        });
    });
</script>
