<script>
        // Simulation logic for UX flow
        document.addEventListener('DOMContentLoaded', () => {
            const sheet = document.getElementById('clientSheet');
            const error = document.getElementById('errorOverlay');

            // Simulate a scan success after 2 seconds
            setTimeout(() => {
                sheet.classList.remove('translate-y-full');
                sheet.classList.add('translate-y-0');
            }, 2000);

            // Simulation function for applying a stamp
            window.applyStamp = function() {
                const undo = document.getElementById('undoAction');
                undo.style.opacity = '1';
                
                // Visual feedback on button
                const btn = event.currentTarget;
                const originalContent = btn.innerHTML;
                btn.innerHTML = '<span class="material-symbols-outlined animate-spin">refresh</span><span>Procesando...</span>';
                btn.disabled = true;

                setTimeout(() => {
                    btn.innerHTML = '<span class="material-symbols-outlined">check_circle</span><span>¡Completado!</span>';
                    btn.classList.replace('bg-[#F5F5F0]', 'bg-secondary');
                    
                    // Reset after few seconds
                    setTimeout(() => {
                        undo.style.opacity = '0';
                        btn.innerHTML = originalContent;
                        btn.classList.replace('bg-secondary', 'bg-[#F5F5F0]');
                        btn.disabled = false;
                        
                        // Optionally hide sheet to simulate flow end
                        // sheet.classList.add('translate-y-full');
                    }, 3000);
                }, 1200);
            };

            // To show error state (manual trigger for demo purposes)
            window.showError = function() {
                error.classList.remove('hidden');
                setTimeout(() => {
                    error.classList.add('hidden');
                }, 4000);
            };
        });
    </script>
